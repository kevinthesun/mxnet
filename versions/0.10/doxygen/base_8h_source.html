269 inline Context Context::FromString(std::string str) {
270  Context ret;
271  try {
272  std::string::size_type l = str.find('(');
273  CHECK_NE(l, std::string::npos);
274  std::string::size_type r = str.find(')');
275  CHECK_EQ(r, str.length()-1);
276 
277  std::string type = str.substr(0, l);
278  int id = std::stoi(str.substr(l+1, r-l-1));
279  if (type == "cpu") {
280  ret = CPU(id);
281  } else if (type == "gpu") {
282  ret = GPU(id);
283  } else if (type == "cpu_pinned") {
284  ret = CPUPinned(id);
285  } else {
286  LOG(FATAL) << "Invalid context string " << str;
287  }
288  } catch (...) {
289  LOG(FATAL) << "Invalid context string " << str;
290  }
291  return ret;
292 }
